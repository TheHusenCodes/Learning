swap two numbers

#include<stdio.h>
int main(){
    int a,b,temp;
    printf("enter number a=:   ");
    scanf("%d",&a);
    printf("enter number b=:   ");
     scanf("%d",&b);
    temp=a;
    a=b;
    b=temp;
    printf("after swapping:a= %d,b= %d ",a,b);
    
    return 0;
}
