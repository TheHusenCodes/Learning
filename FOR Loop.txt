#include <stdio.h>
int main() {
    int n, i, sum = 0;
    printf("Enter n: ");
    scanf("%d", &n);
    
    for(i = 1; i <= n; i++) {
        sum = sum + i;
    }
    
    printf("Sum = %d\n", sum);
    return 0;
}



MULTIPLICATION TABLE
#include <stdio.h>
int main() {
    int n, i;
    printf("Enter a number: ");
    scanf("%d", &n);
    
    for(i = 1; i <= 10; i++) {
        printf("%d x %d = %d\n", n, i, n * i);
    }
    return 0;
}


Factorial of a Number

#include <stdio.h>
int main() {
    int n, i;
    long fact = 1;
    printf("Enter a number: ");
    scanf("%d", &n);
    
    for(i = 1; i <= n; i++) {
        fact = fact * i;
    }
    
    printf("Factorial = %ld\n", fact);
    return 0;
}


Reverse a Number

#include <stdio.h>
int main() {
    int n, rev = 0, rem;
    printf("Enter a number: ");
    scanf("%d", &n);
    
    for(; n != 0; n = n / 10) {
        rem = n % 10;
        rev = rev * 10 + rem;
    }
    
    printf("Reversed number = %d\n", rev);
    return 0;
}
